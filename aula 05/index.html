          import pandas as pd

def remove_outlier(df, column_name):
    # Calcular o primeiro e terceiro quartil (Q1 e Q3)
    Q1 = df[column_name].quantile(0.25)
    Q3 = df[column_name].quantile(0.75)
    
    # Calcular o intervalo interquartílico (IQR)
    IQR = Q3 - Q1
    
    # Definir os limites inferior e superior para os dados aceitáveis
    lower_limit = Q1 - 1.5 * IQR
    upper_limit = Q3 + 1.5 * IQR
    
    # Filtrar os dados para remover os outliers
    df_filtered = df[(df[column_name] >= lower_limit) & (df[column_name] <= upper_limit)]
    
    return df_filtered

# Exemplo de uso
